s.usePlugins=true
s.doPlugins = function(s) {



	s.prop49 = s_getLoadTime();  
	s.eVar49 = "D=c49";



};







/*
 * Plugin: getLoadTime
 */
function s_getLoadTime() {if (!window.s_loadT) {
var e = (new Date).getTime(),
n = window.performance ? performance.timing : 0,
a = n ? n.requestStart : window.inHeadTS || 0;
s_loadT = a ? Math.round((e - a) / 100) : ""
}return s_loadT}
